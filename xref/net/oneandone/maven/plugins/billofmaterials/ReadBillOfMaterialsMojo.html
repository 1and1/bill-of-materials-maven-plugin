16  package net.oneandone.maven.plugins.billofmaterials;
17  
18  import com.google.common.base.Charsets;
19  import com.google.common.io.Files;
20  import java.io.File;
21  import java.io.IOException;
22  import java.util.Locale;
23  import org.apache.maven.plugin.MojoExecutionException;
24  import org.apache.maven.plugin.MojoFailureException;
25  import org.apache.maven.plugins.annotations.LifecyclePhase;
26  import org.apache.maven.plugins.annotations.Mojo;
27  import org.apache.maven.project.MavenProject;
28  
29  /**
30   * Reads a bill of materials for all installed artifacts and puts it's content in a property
31   * called <tt>qaBillOfMaterials</tt> of the project which may be reused by templating plugins.
32   *
33   * Spits out a warning when the file does not exist. For usage see the integration tests.
34   *
35   * <p>This in the standard format for the <tt>sha1sum</tt> command including meta information:</p>
36   * <pre>
37   * # company:company-parent-pom:1.0-SNAPSHOT user=mirko
38   * 2dcb20b977ff170dd802c30b804229264c97ebf6  company-parent-pom-1.0-SNAPSHOT.pom
39   * # company:child1:1.0-SNAPSHOT user=mirko
40   * ed5b932c3157b347d0f7a4ec773ae5d5890c1ada  child1-1.0-SNAPSHOT-sources.jar
41   * 8294565e2a5d99b548b111fe6262719331436143  child1-1.0-SNAPSHOT.jar
42   * 082fa2206c4a00e3f428e9100199a0337ad42fdb  child1-1.0-SNAPSHOT.pom
43   * # company:child2:1.0-SNAPSHOT user=mirko
44   * 05d419cf53e175c6e84ddc1cf2fccdc9dd109c6b  child2-1.0-SNAPSHOT-sources.jar
45   * df633b963220ba124ffa80eb6ceab676934bb387  child2-1.0-SNAPSHOT.jar
46   * 5661e9270a02c5359be47615bb6ed9911105d878  child2-1.0-SNAPSHOT.pom
47   * </pre>
48   *
49   * @author Mirko Friedenhagen &lt;mirko.friedenhagen@1und1.de&gt;
50   */
51  @Mojo(name = "read", defaultPhase = LifecyclePhase.INSTALL)
52  public class ReadBillOfMaterialsMojo extends AbstractBillOfMaterialsMojo {
53  
54      /**
55       * Default constructor for maven.
56       */
57      ReadBillOfMaterialsMojo() {
58          super();
59      }
60  
61      /**
62       * Just for tests.
63       * @param billOfMaterialsPath path to bom.
64       * @param project current project
65       */
66      ReadBillOfMaterialsMojo(File billOfMaterialsPath, MavenProject project) {
67          super(billOfMaterialsPath, project);
68      }
69  
70      @Override
71      public void execute() throws MojoExecutionException, MojoFailureException {
72          final File bomFile = calculateBillOfMaterialsFile();
73          getLog().info("Reading bill of materials from " + bomFile);
74          try {
75              final String qaBillOfMaterials = Files.toString(bomFile, Charsets.UTF_8);
76              getProject().getProperties().put("qaBillOfMaterials", qaBillOfMaterials);
77          } catch (IOException e) {
78              getLog().warn(String.format(
79                          Locale.ENGLISH, "Could not read content '%s', did you run bill-of-materials:create?", e));
80          }
81      }
82  
83  }
